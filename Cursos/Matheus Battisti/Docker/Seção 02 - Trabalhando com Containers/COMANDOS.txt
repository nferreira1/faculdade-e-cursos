CRIAR E EXECUTAR UM CONTAINER
docker run NOME_DO_CONTAINER

CRIAR, EXECUTAR E NOMEAR UM CONTAINER
docker run --name NOME_DO_CONTAINER NOME_QUE_DESEJA_DAR_AO_CONTAINER

EXECUTAR UM CONTAINER COM INTERAÇÃO, OU SEJA, ELE FICARÁ EXECUTANDO NO TERMINAL
docker run -it ubuntu

EXECUTAR UM CONTAINER EM BACKGROUND
docker run -d 

EXECUTAR UM CONTAINER E EXPORT UMA PORTA DO CONTAINER
docker run -p PORTA_QUE_ESTA_SENDO_EXPOSTA_NA_MAQUINA:PORTA_QUE_ESTA_RECEBENDO_DO_CONTAINER

EXECUTAR UM CONTAINER JÁ CRIADO
docker start NOME_DO_CONTAINER OU ID_DO_CONTAINER

PARAR UM CONTAINER
docker stop NOME_DO_CONTAINER OU ID_DO_CONTAINER

LISTAR CONTAINERS QUE ESTÃO EXECUTANDO
docker ps OU docker container ls

LISTAR CONTAINERS QUE NÃO ESTÃO RODANDO
docker ps -a OU docker container ls -a

LISTAR IMAGENS QUE ESTÃO INSTALADAS
docker image ls

VER LOGS DE UM CONTAINER
docker logs NOME_DO_CONTAINER OU ID_DO_CONTAINER (PODE-SE USAR O -f PARA FICAR SEMPRE EXIBINDO OS LOGS)

REMOVER CONTAINER
docker rm NOME_DO_CONTAINER OU ID_DO_CONTAINER
